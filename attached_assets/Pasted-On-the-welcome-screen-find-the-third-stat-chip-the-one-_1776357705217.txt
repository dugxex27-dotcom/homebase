Add this JavaScript — place it just before the closing </body> tag or inside your existing JS file:
javascript// Referral info modal toggle
(function() {
  const trigger = document.getElementById('mhb-referral-trigger');
  const modal   = document.getElementById('mhb-referral-modal');
  const close   = document.getElementById('mhb-referral-close');

  if (!trigger || !modal) return;

  // Toggle on tap/click
  trigger.addEventListener('click', function(e) {
    e.stopPropagation();
    const isOpen = modal.style.display === 'block';
    modal.style.display = isOpen ? 'none' : 'block';
  });

  // Close button
  if (close) {
    close.addEventListener('click', function(e) {
      e.stopPropagation();
      modal.style.display = 'none';
    });
  }

  // Close when tapping anywhere else on the screen
  document.addEventListener('click', function() {
    modal.style.display = 'none';
  });

  // Hover support for desktop
  trigger.addEventListener('mouseenter', function() {
    modal.style.display = 'block';
  });

  trigger.closest('.mhb-stat-chip').addEventListener('mouseleave', function() {
    modal.style.display = 'none';
  });
})();
Do not change any other tiles, any other screens, or any existing functionality.
