Step 1 — Understand the architecture shift
We are moving from a React SPA to server-rendered HTML pages. The Express server will serve each HTML file directly. Real data from the database will be injected into each page using template literal replacement — find placeholder tokens like {{USER_NAME}}, {{HWS_SCORE}}, etc. in the HTML and replace them with real values before sending the response. Do not use a template engine like Handlebars or EJS — just use Node's built-in string replacement on the HTML file content.
Step 2 — Add data injection helper to server.js
Add this helper function to server.js:
javascriptasync function serveInjected(res, filename, data = {}) {
  const fs = require('fs');
  const path = require('path');
  let html = fs.readFileSync(path.join(__dirname, 'public', filename), 'utf8');
  for (const [key, value] of Object.entries(data)) {
    html = html.replaceAll(`{{${key}}}`, value ?? '');
  }
  res.setHeader('Content-Type', 'text/html');
  res.send(html);
}
