```javascript
// ============================================
// google-auth.js
// Create this as a new file and import it
// into your main app file
// ============================================

// ---- CONFIG ----
// Replace with your actual Client ID
// or pull from environment: process.env.GOOGLE_CLIENT_ID
const GOOGLE_CLIENT_ID = '309382820440-3gv5nugurvbo5gf767fdj6o38dq43thk.apps.googleusercontent.com';

// ---- ROUTES ----
// Update these to match your actual dashboard URLs
const DASHBOARD_ROUTES = {
  homeowner:  '/dashboard/homeowner',
  contractor: '/dashboard/contractor',
  agent:      '/dashboard/agent'
};

// ---- INITIALIZE ----
// Call this once when the welcome screen loads
function initGoogleSignIn() {
  if (typeof google === 'undefined') {
    console.error('Google Sign-In library not loaded.');
    return;
  }

  google.accounts.id.initialize({
    client_id: GOOGLE_CLIENT_ID,
    callback: handleGoogleCredential,
    auto_select: false,
    cancel_on_tap_outside: true,
  });
}

// ---- TRIGGER ----
// Called when user taps "Continue with Google"
// Pass the user type: 'homeowner', 'contractor', or 'agent'
function handleGoogleSignIn(userType) {
  if (typeof google === 'undefined') {
    showAuthError();
    return;
  }

  // Store user type so we know where to route after auth
  sessionStorage.setItem('mhb_user_type', userType);

  // Show Google's sign-in popup
  google.accounts.id.prompt((notification) => {
    if (notification.isNotDisplayed() || notification.isSkippedMoment()) {
      // Popup was blocked or dismissed — fall back to redirect flow
      triggerGoogleRedirect(userType);
    }
  });
}

// ---- FALLBACK ----
// If popup is blocked use redirect flow instead
function triggerGoogleRedirect(userType) {
  const client = google.accounts.oauth2.initCodeClient({
    client_id: GOOGLE_CLIENT_ID,
    scope: 'email profile openid',
    ux_mode: 'redirect',
    redirect_uri: `${window.location.origin}/auth/google/callback`,
    state: userType,
  });
  client.requestCode();
}

// ---- CALLBACK ----
// Google calls this with the credential after sign-in
async function handleGoogleCredential(response) {
  const userType = sessionStorage.getItem('mhb_user_type') || 'homeowner';

  try {
    // Send the Google credential token to your backend
    // Your backend should verify it and create/find the user
    const result = await fetch('/auth/google', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        credential: response.credential,
        user_type: userType,
      }),
    });

    if (!result.ok) throw new Error('Auth failed');

    const data = await result.json();

    // Clear the stored user type
    sessionStorage.removeItem('mhb_user_type');

    // Route to the correct dashboard
    routeAfterSignIn(userType, data);

  } catch (error) {
    console.error('Google auth error:', error);
    showAuthError();
  }
}

// ---- ROUTING ----
function routeAfterSignIn(userType, data) {
  const route = DASHBOARD_ROUTES[userType] || '/dashboard';

  // If using React Router:   navigate(route)
  // If using Vue Router:     router.push(route)
  // If using plain JS:
  window.location.href = route;
}

// ---- ERROR ----
function showAuthError() {
  // Replace this with however your app shows errors
  const errorEl = document.getElementById('mhb-auth-error');
  if (errorEl) {
    errorEl.style.display = 'block';
    setTimeout(() => { errorEl.style.display = 'none'; }, 4000);
  }
}

// ---- ON LOAD ----
// Initialize when the page is ready
document.addEventListener('DOMContentLoaded', initGoogleSignIn);
```
